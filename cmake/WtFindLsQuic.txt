IF(WIN32)
  SET(LSQUIC_INCLUDE_DIR "")
ELSEIF(NOT APPLE)
  FIND_PATH(LSQUIC_INCLUDE_DIR
    NAMES
      /liburing
    PATHS
      /usr/include
      /usr/local/include
  )
IF(LSQUIC_INCLUDE_DIR)
  SET(LSQUIC_INCLUDE_DIR ${LSQUIC_INCLUDE_DIR})
ENDIF(LSQUIC_INCLUDE_DIR)
ENDIF(WIN32)


IF(WIN32)
  SET(LSQUIC_LIBRARY lsquic)
ELSEIF(APPLE)
  FIND_LIBRARY(LSQUIC_LIBRARY NAMES lsquic)
ELSEIF(UNIX)
  FIND_LIBRARY(LSQUIC_LIBRARY NAMES lsquic)
  FIND_LIBRARY(LSQUIC_LIBRARY
    NAMES
      lsquic
    PATHS
      ${LSQUIC_INCLUDE_DIR}/../
      /usr/lib
      /usr/lib64
      /usr/local/lib
      /opt/local/lib
  )
  IF(LSQUIC_LIBRARY)
    SET(LSQUIC_LIBRARY ${LSQUIC_LIBRARY})
  ENDIF(LSQUIC_LIBRARY)
ENDIF(WIN32)



SET(LSQUIC_FOUND FALSE)

IF(LSQUIC_LIBRARY AND LSQUIC_INCLUDE_DIR)
  SET(LSQUIC_FOUND TRUE)
ENDIF(LSQUIC_LIBRARY AND LSQUIC_INCLUDE_DIR)
