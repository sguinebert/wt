IF(WIN32)
  SET(URING_INCLUDE_DIR "")
ELSEIF(NOT APPLE)
  FIND_PATH(URING_INCLUDE_DIR
    NAMES
      /liburing
    PATHS
      /usr/include
  )
IF(URING_INCLUDE_DIR)
  SET(URING_INCLUDE_DIR ${URING_INCLUDE_DIR})
ENDIF(URING_INCLUDE_DIR)
ENDIF(WIN32)


IF(WIN32)
  SET(URING_LIBRARY uring)
ELSEIF(APPLE)
  FIND_LIBRARY(URING_LIBRARY NAMES uring)
ELSEIF(UNIX)
  FIND_LIBRARY(URING_LIBRARY NAMES uring)
  IF(URING_LIBRARY)
    SET(URING_LIBRARY ${URING_LIBRARY})
  ENDIF(URING_LIBRARY)
ENDIF(WIN32)
